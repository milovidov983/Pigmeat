347  static void ServeBuild(object source, FileSystemEventArgs e)
348  {
349  var RelativePath = Path.GetRelativePath("./", e.FullPath); // Relative paths are nicer to work with
350  /*
351  Filter out the following situations:
352  1. Changes by Pigmeat iself
353  2. Changes recorded by git
354  3. Changes to directories (no purpose in doing anything when this occurs)
355  4. Changes made mere milliseconds apart
356  */
357  if(File.GetAttributes(e.FullPath).HasFlag(FileAttributes.Directory) || Path.GetFileName(RelativePath).Equals("_global.yml") || RelativePath.Contains("output/") || RelativePath.Contains(".git/") || DateTime.Now.Subtract(LastFileWatcherEvent).TotalMilliseconds < 500)
358  {
359  return;
360  }
361  LastFileWatcherEvent = DateTime.Now; // Keeps track of when the last proper event occured
362 
363  Console.WriteLine("\n" + RelativePath + " has been modified. Rebuilding ");
364 
365  JObject Global = JObject.Parse(IO.GetGlobal());
366  string[] IncludedFiles = JsonConvert.DeserializeObject<string[]>(Global["included-files"].ToString());
367  string[] IncludedPages = JsonConvert.DeserializeObject<string[]>(Global["included-pages"].ToString());
368 
369  // If a layout is changed, overwrite it in the cache and rebuild its dependencies
370  if(Path.GetDirectoryName(RelativePath).Equals("layout"))
371  {
372  IO.GetLayoutContents(RelativePath, true);
373  foreach(var file in Directory.GetFiles("./", "*", SearchOption.TopDirectoryOnly))
374  {
375  if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
376  {
377  JObject PageObject = Page.GetPageObject(RelativePath);
378  if(PageObject["layout"].ToString().Equals(Path.GetFileNameWithoutExtension(RelativePath)))
379  {
380  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
381  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
382  }
383  }
384  }
385  }
386  // If a snippet is changed, rebuild its dependencies
387  else if(Path.GetDirectoryName(RelativePath).Equals("snippets"))
388  {
389  foreach(var file in Directory.GetFiles("./", "*", SearchOption.TopDirectoryOnly))
390  {
391  if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
392  {
393  if(File.ReadAllText(file).Contains("{! snippet " + Path.GetFileName(RelativePath)))
394  {
395  JObject PageObject = Page.GetPageObject(RelativePath);
396  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
397  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
398  }
399  }
400  }
401  }
402  // If a document is changed, rebuild it
403  else if(Path.GetExtension(RelativePath).Equals(".md") || Path.GetExtension(RelativePath).Equals(".html"))
404  {
405  JObject PageObject = Page.GetPageObject(RelativePath);
406  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
407  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
408  }
409  // If the stylesheet(s) are changed, rebuild them
410  else if(Path.GetExtension(RelativePath).Equals(".scss") || Path.GetExtension(RelativePath).Equals(".sass"))
411  {
412  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
413  File.WriteAllText("./output/" + Path.GetDirectoryName(RelativePath) + "/" + Path.GetFileNameWithoutExtension(RelativePath) + ".css", Scss.ConvertToCss(File.ReadAllText(RelativePath)).Css);
414  }
415  // If a file has changed, copy it again
416  else if(!Path.GetExtension(RelativePath).Equals(".json") && !Path.GetExtension(RelativePath).Equals(".yml"))
417  {
418  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
419  File.Copy(RelativePath, "./output/" + RelativePath, true);
420  }
421  // If 'included' and changed, then re-output
422  try
423  {
424  foreach(var file in IncludedFiles)
425  {
426  if(file.Equals(RelativePath))
427  {
428  Directory.CreateDirectory("./output/" + Path.GetDirectoryName(RelativePath));
429  try
430  {
431  File.Copy(RelativePath, "./output/" + RelativePath, true);
432  }
433  catch(IOException)
434  {
435  IO.IncludeDirectory(RelativePath, "./output/" + RelativePath);
436  }
437  }
438  }
439  }
440  catch(Exception ex)
441  {
442  Console.WriteLine(ex);
443  }
444  try
445  {
446  foreach(var file in IncludedPages)
447  {
448  if(file.Equals(RelativePath))
449  {
450  JObject PageObject = Page.GetPageObject(RelativePath);
451  Directory.CreateDirectory(Path.GetDirectoryName("./output/" + PageObject["url"].ToString()));
452  File.WriteAllText("./output/" + PageObject["url"].ToString(), PageObject["content"].ToString());
453  }
454  }
455  }
456  catch(Exception ex)
457  {
458  Console.WriteLine(ex);
459  }
460  return;
461  }
